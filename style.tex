% ============================================================
% Formattazione Titolo Capitolo (Stile Display)
% ============================================================

% 1. Forza lo stile "prefisso" (mette il numero sopra il titolo)
\KOMAoption{chapterprefix}{true}

% 2. Centra tutto l'intestazione del capitolo
\let\raggedchapter\centering

% 3. Definisce lo spazio tra "Capitolo X" e il "Titolo"
% Usiamo \renewcommand perché KOMA ha un'impostazione specifica per questo
\renewcommand*{\chapterheadmidvskip}{\vskip 15pt}

% 4. Configura le distanze (Sopra e Sotto)
\RedeclareSectionCommand[
  beforeskip=-30pt,
  afterskip=40pt
]{chapter}

% 5. Opzionale: Se vuoi personalizzare la scritta "Chapter 1" 
% e renderla diversa dal titolo (es. più piccola o meno grassetta)
\setkomafont{chapterprefix}{\normalfont\small\scshape}
\setkomafont{chapter}{\normalfont\Large\bfseries\scshape}

\setkomafont{section}{\normalfont\Large\bfseries\scshape}

%% ============================================================
% Definizione Ambiente Axiom (Versione Fixata)
% ============================================================

\theoremstyle{plain}
\newtheorem{theorem}{\normalfont\bfseries\scshape Theorem}[section]
\newtheorem{proposition}[theorem]{\normalfont\bfseries\scshape Proposition}

\newtheoremstyle{myplain}% Custom style for theorems
    {1ex}% Space above
    {1ex}% Space below
    {\itshape}% Body font
    {}% Indent amount
    {\bfseries}% Theorem head font
    {.}% Punctuation after theorem head
    {0.5em}% Space after theorem head
    {}% Theorem head spec

\newtheoremstyle{mydefinition}% Custom style for definitions and similar environments
    {1ex}% Space above
    {1ex}% Space below
    {\normalfont}% Body font
    {}% Indent amount
    {\bfseries}% Theorem head font
    {.}% Punctuation after theorem head
    {0.5em}% Space after theorem head
    {}% Theorem head spec

\theoremstyle{myplain}
\newtheorem{lemma}[theorem]{\normalfont\bfseries\scshape Lemma}
\newtheorem{corollary}[theorem]{\normalfont\bfseries\scshape Corollary}

\theoremstyle{mydefinition}
\newtheorem{definition}[theorem]{\normalfont\bfseries\scshape Definition}
\newtheorem{example}[theorem]{Example}
\newtheorem{remark}[theorem]{Remark}

\renewcommand{\thetheorem}{\arabic{theorem}}
