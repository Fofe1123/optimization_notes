\begin{definition}[Convex function]
    A function $f: \Rn \to \R$ is convex if its domain is a convex set and for all $x, y$ in the domain and all $\theta \in [0, 1]$,
    \[
        f(\theta x + (1 - \theta) y) \leq \theta f(x) + (1 - \theta) f(y).
    \]
\end{definition}
