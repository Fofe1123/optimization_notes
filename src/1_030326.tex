\section{A quick summary of calculus 1 \& 2}

\begin{definition}[Norm/scalar product]
    
    Given $x = (x_1, \dots, x_n) \in \Rn$ the Euclidian norm (or $\ell^2$ norm) is defined as 

    \[ \norm{x} = \left(\sum_{i = 1}^n x_i^2\right)^{1/2}. \]

    For $x,y \in \Rn$ the \emph{scalar product} is given by 

    \[ \langle x,y \rangle = x \cdot y = x^\top y = \sum_{i = 1}^n x_i y_i. \] 

    Moreover, the scalar product induces a norm. Indeed, one can verify that 

    \[ \norm{x} = \sqrt{\langle x,x\rangle}. \]

\end{definition}
