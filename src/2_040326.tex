Our main focus from now on is to solve the problem 

\[ \inf_{x \in A} f(x) \]

for some $A \subseteq \Rn$ and some $f : \Rn \to \R$. We will restrict ourselfs only to certain functions and sets.

\begin{definition}[Infimum/supremum]
    
    Let $E \subseteq \R$. We denote with $\inf E =$ largest $z \in \R$ such that $z \le x$ for every $x \in E$. By convention we set $\inf \emptyset = +\infty$. 

    Similarly, we denote with $\sup E$ the smallest $z \in \R$ such that $z \ge x$ for every $x \in E$, with the convention $\sup \emptyset = -\infty$. More formally, 

    \begin{align*}
        z &= \inf E \quad \text{ if } z \le x \,\forall x \in E \text{ and } \forall \eps > 0\, \exists y \in E \text{ s.t. } y \le z + \eps \\
        z &= \sup E \quad \text{ if } z \ge x \,\forall x \in E \text{ and } \forall \eps > 0\, \exists y \in E \text{ s.t. } y \ge z - \eps 
    \end{align*}

\end{definition}

\begin{remark}
    
    The $\inf$ and $\sup$ aren't necessarely inside the set. If they are, we speak of $\min$ and $\max$.

\end{remark}

\begin{axiom}{Completeness of $\mathbb{R}$}{completeness}
    Every bounded sequence $\set{x_n}_{n \in \N} \subseteq \R$ admits a supremum  in $\R$.
\end{axiom}

\begin{theorem}[Bolzano-Weierstrass]

    Any bounded sequence $\set{x_n}_{n \in \N} \subseteq \R$ admits a cluster point, i.e. there exists $\bar{x} \in \R$ and a subsequence $\set{x_{n_j}}_{j \in \N}$ such that $x_{n_j} \to \bar{x}$ as $j \to +\infty$.
    
\end{theorem}

\begin{proof}
    
    Since $\set{x_n}_n$ is bounded, there exists $R > 0$ such that $x_n \in [-R,R]$ for every $n \in \N$. Set $I_1 = [-R,R]$, select $x_{n_1}$ a value in $I_1$. Divide $I_1$ in half and select the one subinterval containing inifitely many $x_n$s, call it $I_2 = [a_2,b_2]$ (here $a_2 = -R, b_2 = 0$ or $a_2 = 0, b_2 = R$). We have $I_2 \subset I_1$, $\abs{I_2} = \abs{I_1}/2$, and choose $x_{n_2}$ a value in $I_2$. Repeat this process to cut and select. We end up, at the $k$-th step, with a sequence of subintervals $I_1,I_2,\dots,I_k$ and 3 subsequences: $x_{n_1},x_{n_2}, \dots, x_{n_k}$ the sequence of selected values, $a_1,a_2,\dots,a_k$ the sequence of left boundaries and $b_1,b_2,\dots,b_k$ the sequence of right boundaries.

    We observe that $a_{j+1} \ge a_j$ and $b_{j+1} \le b_j$ for all $1 \le j \le k$. Moreover, $a_j \le R$ and $b_j \ge -R$ for all $j$, so, by the axiom of completeness of $\R$, 

    \begin{itemize}
        \item there exists $\bar{a} \in \R$ such that $a_k \to \bar{a} = \displaystyle\sup_k a_k$
        \item there exists $\bar{b} \in \R$ such that $b_k \to \bar{b} = \displaystyle\inf_k b_k$ 
    \end{itemize}

    We observe that $\bar{a} = \bar{b}$. Indeed, $\abs{\bar{b} - \bar{a}} = \abs{\bar{b} - b_k + b_k - a_k + a_k -\bar{a}} \le \abs{\bar{b} - b_k} + \abs{\bar{a} - a_k} + \abs{b_k - a_k}$. The terms $\abs{\bar{b} - b_k}$ and $\abs{\bar{a} - a_k}$ goes to 0 as $k \to +\infty$ for what we observed before. Moreover, $\abs{b_k - a_k} = 2R/2^{k-1} \to 0$ as $k \to +\infty$. 

    Set $\bar{x} = \bar{a} (= \bar{b})$. Then the subsequence $\set{x_{n_j}}_j$ is converging to $\bar{x}$: indeed $ \abs{x_{n_j} - \bar{x}} = \abs{x_{n_j} - a_j + a_j - \bar{x}} \le \abs{a_{n_j} - a_j} + \abs{a_j - \bar{x}}$.

    Since $a_j \to \bar{x} = \bar{a}$, the term $\abs{a_j - \bar{x}} \to 0$. Notice that, by construction, $x_{n_j} \in I_j$, so $\abs{x_{n_j} - a_j} \le \abs{I_j} \to 0$.

\end{proof}

\begin{definition}
    
    Let $f : \Rn \to \R$. We say that $\bar{x} \in \Rn$ is a \emph{local maximum} for $f$ if there exists a neighbourhood $U$ of $\bar{x}$ such that $f(x) \le f(\bar{x})$ for every $x \in U$. A \emph{local strict maximum} has the strict inequality.

    Similarly, $\bar{x}$ is \emph{local minumum} for $f$ if $\exists U$ neighbourhood of $\bar{x}$ such that $f(x) \ge f(\bar{x})$ for all $x \in U$.

    If there exists $\bar{x}$ such that $f(x) = \displaystyle \inf_{x \in A} f(x)$, the $\bar{x}$ is a \emph{global minimum} over $A$.

\end{definition}

\begin{definition}
    
    We denote with $\arg_{x \in A}\min f(x)$ the set of minumum points for $f$ over $A$, i.e 

    \[ \arg_{x \in A}\min f(x) = \set{\bar{x} \in \Rn : f(\bar{x}) = \inf_{x \in A} f(x)}. \]

\end{definition}

\begin{theorem}[Weierstrass]

    Let $A \subseteq \Rn$ be a compact set, $f : A \to \R$ a continous function. The there exists $x_m$, $x_M \in A$ such that
    
    \begin{align*}
        f(x_m) &= \min_{x \in A} f(x) \\
        f(x_M) &= \max_{x \in A} f(x).
    \end{align*}
    
\end{theorem}

\begin{proof}
    
    We prove the statement for the minimum point. 

    Let $\ell = \displaystyle\inf_{x \in A} f(x)$. We can always find a sequence $\set{x_n}_n \subseteq A$ such that $f(x_n) \to \ell$ by the continuity of $f$. Since $A$ is bounded, $\set{x_n}_n$ is bounded as well. By Bolzano-Weierstrass, there exists a subsequence $\set{x_{n_k}}_k$ converging to $\bar{x} \in A$ by closerness of $A$. By continuity, in particular $f$ is continous at $\bar{x}$, 

\end{proof}

\begin{remark}
    
    For $f : \Rn \to \R$, we define the \emph{gradient} of $f$ as 

    \[ \nabla f(c) = \left( \frac{\partial f}{\partial x_1}(c), \dots, \frac{\partial f}{\partial x_n}(c) \right) \]

    the vector of partial derivatives, defined as

    \[ \frac{\partial f}{\partial x_i}(c) = \lim_{h \to 0} \frac{f(c + he_i) - f(c)}{h} \]

    whenever this limit exists finite, and where $e_i$ is the $i$-th vector of the canonical base of $\Rn$. Furthermore, we say that $f$ is \emph{differentiable}   at $c$ if

    \[ f(x) = f(c) + \langle \nabla f(c), x - c \rangle + \mathcal{O}(\norm{x - c}^2) \]

    for $x \to c$.

\end{remark}

\begin{definition}[Convex set]

    Let $C \subseteq \Rn$. We say that $C$ is \emph{convex} if, for any $x,y \in C$, and for any $\alpha \in [0,1]$, $\alpha x + (1-\alpha)y \in C$.
    
\end{definition}

\begin{proposition}
    
    \begin{enumerate}
        \item[a)] $\displaystyle\bigcap_{i \in I} C_i$ is convex if $C_i$ is convex for any $i \in I$ finite or countable;
        \item[b)] the \emph{algebraic sum} $C_1 + C_2 = \set{x + y : x \in C_1,\, y \in C_2}$ is convex if $C_1$ and $C_2$ are convex;
        \item[c)] the set $\lambda C = \set{\lambda x : x \in C}$ is convex if $C$ is convex for any $\lambda \in \R$;
        \item[d)] $\overline{C},C^\circ$ are convex if $C$ is convex;
        \item[e)] if a function $f : C \to \R$ is affine, then the set $f(C) = \set{y : y = f(x), \, x \in C}$ is convex if $C$ is convex.
    \end{enumerate}

\end{proposition}

\begin{proof}
    
    We prove all the propositions separately.

    \paragraph{\underline{a)}:} suppose $C_i$ convex and let $C = \displaystyle\bigcap_{i \in I} C_i$. Let $x,y \in C$, $\alpha \in [0,1]$. Since $x,y \in C$, in particular $x,y \in C_i$ for all $i \in I$. But then also $\alpha x + (1-\alpha)y \in C_i$ for all $i \in I$, hence $C$ is convex.

    \paragraph{\underline{b)}:} let $x = x_1 + x_2 \in C_1 + C_2$, $x_1 \in C_1$ and $x_2 \in C_2$, take $y \in C_1 + C_2$ of the same form. 
    
    \[ \alpha x + (1-\alpha y) = \underbrace{\alpha x_1 + (1-\alpha)y_1}_{\in C_1} + \underbrace{\alpha x_2 + (1-\alpha)y_2}_{\in C_2} \in C. \]

    \paragraph{\underline{c)}:} trivial.

    \paragraph{\underline{d)}:} consider $\set{x_k}_k \subseteq C$ and $\set{y_k}_k \subseteq C$ such that $x_k \to x$ and $y_k \to y$. Then the sequence $\set{\alpha x_k + (1-\alpha)y_k}_k \subseteq C$. But then also $\alpha x + (1-\alpha)y \in C$ as limit of sequences valued in $C$. 

    Let now $x,y \in C^\circ$. Then there exists $r > 0$ such that $B(x,r) \subseteq C$ and $B(y,r) \subseteq C$. Consider $\alpha x + (1-\alpha)y$ . Our claim is: $B(\alpha x + (1-\alpha)y, r) \subseteq C$. Indeed, any $z \in B(\alpha x + (1-\alpha)y,r)$ is of the form $z = \alpha x + (1-\alpha)y + w$ with $\norm{w} < r$, so $\alpha (x+w) + (1-\alpha)(y+w) \in C$ by convexity of $C$, hence the claim.

\end{proof}
